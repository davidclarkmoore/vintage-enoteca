 15 /**
 16  * Creates a new location.
 17  * @class LocatorLocation represents a location.
 18  * @param {Document} xml The XML Object containing the location's data.
 19  * @param {Number} index The unique index for the location.
 20  */
 21 var LocatorLocation = function(xml, index){
 22 	/**
 23 	 * {Document} Reference to the XML object that has this location's data.
 24 	 * The XML object is referenced directly to save on memory and time.
 25 	 */
 26 	this.xml = xml;
 27 	/**
 28 	 * {Number} The unique index of the location.
 29 	 */
 30 	this.index = index;
 31 	/**
 32 	 * {GMarker} A reference to the <a href="http://code.google.com/apis/maps/documentation/reference.html#GMarker">GMarker</a> object representing this location on the <a href="http://code.google.com/apis/maps/documentation/reference.html#GMap2">GMap2</a>.
 33 	 */
 34 	this.marker = null;
 35 	/**
 36 	 * {GLatLng} A reference to the <a href="http://code.google.com/apis/maps/documentation/reference.html#GLatLng">GLatLng</a> object for this location (its latitude and longitude).
 37 	 */
 38 	this.point = null;
 39 	/**
 40 	 * {Number} The location's most recently calculated distance.
 41 	 */
 42 	this.distance = null;
 43 	/**
 44 	 * {Object} Reference to the DIV which holds the content for this location's small info window.
 45 	 */
 46 	this.smallHTML = null;
 47 	/**
 48 	 * {Object} Reference to the DIV which holds the content for this location's large info window.
 49 	 */
 50 	this.largeHTML = null;
 51 	/**
 52 	 * {GInfoWindowOptions} Info window options for this location (<a href="http://code.google.com/apis/maps/documentation/reference.html#GInfoWindowOptions">GInfoWindowOptions</a>).
 53 	 */
 54 	this.infoWindowOptions = {};
 55 };

 63 LocatorLocation.prototype.getAttribute = function(attribute){

 72 LocatorLocation.prototype.getAttributeAsNumber = function(attribute){

 76 /**
 77  * Returns the value of a nested child element of the location XML data.
 78  * @return {String} The data value of as a string.
 79  */
 80 LocatorLocation.prototype.getData = function(){
 81 	var args = Array.prototype.slice.call(arguments);
 82 	
 83 	return $('> ' + args.join('>'), this.xml).text();
 84 };
 85 
 86 /**
 87  * Returns the value of a nested child element of the location XML data as a Number.
 88  * @return {Number} The data value as a Number.
 89  */
 90 LocatorLocation.prototype.getDataAsNumber = function(){
 91 	return Number(this.getData.apply(this, arguments));
 92 };

 94 /**
 95  * Returns the attribute value of a nested child element of the location XML data.
 96  * @return {String} The data attribute value.
 97  */
 98 LocatorLocation.prototype.getDataAttribute = function(){
 99 	// Turn the arguments object into a normal array.
100 	var args = Array.prototype.slice.call(arguments);
101 	// The last argument is the attribute name.
102 	var attribute = args.pop();
103 	return $('> ' + args.join('>'), this.xml).attr(attribute);
104 };

134 	}

561 	/**
562 	 * Processes the locator data to create compiled templates and adds the states to the map overlay.
563 	 * Executes the important methods {@link Locator.initialize} and {@link Locator.reset} to get the locator
564 	 * in a valid state. When complete, the {@link Locator.addStatesToOverlay} function is executed with {@link Locator.loadLocations}
565 	 * as a parameter so it is executed afterward.
566 	 * @param {Document} data XML document containing the data to be processed.
567 	 */
568 	processData: function(data){
569 		Locator.xml.data = data;
570 		
571 		// Loop through and create the templates.
572 		$('data > templates > template', data).each(function(){
573 			Locator.templates[$(this).attr('name')] = $.createTemplate($(this).text(), null, {filter_data: false});
574 		});
575 		
576 		// Grab the sidebar content and remove it.
577 		// The sidebar is removed because the locator HTML already includes
578 		// a <div> with id of "sidebar".
579 		Locator.sidebarInitialContent = $('#sidebar').html();
580 		$('#sidebar').remove();
581

831 	 * @param {Number} distance Maximum number of miles to search within.
832 	 * @return {LocatorLocation[]} An array of location objects sorted ascending by distance.

834 	findLocationsNearLatLng: function(latlng, distance){
835 		var sortedLocations = [];

848 			}
849

853 			location.distance = Locator.utilities.metersToMiles(latlng.distanceFrom(new GLatLng(location.getAttributeAsNumber('lat'), location.getAttributeAsNumber('lng'))));
854 			
855 			if (location.distance <= distance)
856 				sortedLocations.push(location);
857 		}
858 		
859 		// Sort the locations by distance.
860 		sortedLocations.sort(Locator.sortLocationsByDistance);
861 		
862 		return sortedLocations;
