955 		/**
956 		 * Executed as a callback to maps.google.Geocoder.geocode or can be executed independently
957 		 * if a proper maps.google.GeocodeResponse object is passed as the first parameter and
958 		 * google.maps.GeocoderStatus.OK is passed as the second parameter.
959 		 * Determines whether the searched address is inside, or within the buffer distance, of
960 		 * the currently active distribution and displays {@link ServiceAreas.searchInfoWindow}
961 		 * with a message determined by the address searched and service availability.
962 		 * @param {google.maps.GeocoderResult[]} results An array of geocoder results.
963 		 * @param {google.maps.GeocoderStatus} status The status of the results.
964 		 * @memberOf ServiceAreas
965 		 * @name checkLocation
966 		 */
967 		this.checkLocation = function(results, status){
968 			var latlng = null, distance = 65535, message = null, text = '', distribution = null;
969 			
970 			// If the response is valid, then create a LatLng object from its coordinates.
971 			if (status == google.maps.GeocoderStatus.OK && results.length == 1) {
972 				_searchHistory.push(results);
973 				latlng = results[0].geometry.location;
974 			}
975 			
976 			if (latlng != null) {
977 				// Show the marker at the location.
978 				_self.searchMarker.setPosition(latlng);
979 				_self.searchMarker.setVisible(true);
980 				
981 				distribution = _self.currentDistribution;
982 				message = distribution.getMessageFromAddress(results[0].formatted_address);
983 				text = message.getOutsideMessage();
984 				
985 				// Check if if distribution contains the location and, if not, find the
986 				// distance to it from the distribution area.
987 				if (distribution.containsLatLng(latlng))
988 					distance = 0;
989 				else {
990 					distance = distribution.distanceToLatLng(results[0].formatted_address, latlng);
991 				}
992 				
993 				// Set the message depending on the distance.
994 				if (distance == 0)
995 					text = message.getInsideMessage();
996 				else if (distance > 0 && distance <= _self.config.bufferDistance)
997 					text = message.getEdgeMessage();
998 				
999 				text = '<p><strong>' + results[0].formatted_address + '</strong></p>' + text;
1000 				
1001 				if (_self.currentInfoWindow)
1002 					_self.currentInfoWindow.close();
1003 					
1004 				_self.currentInfoWindow = _self.searchInfoWindow;
1005 				_self.searchInfoWindow.setContent(text);
1006 				_self.searchInfoWindow.open(_map, _self.searchMarker);
1007 			}
1008 		};
